cmake_minimum_required(VERSION 3.20)
project(microgpt-mine VERSION 1.0 LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 23)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF) # Disables compiler-specific hacks

set(CMAKE_EXPORT_COMPILE_COMMANDS ON)

find_package(fmt CONFIG REQUIRED)
find_package(catch2 CONFIG REQUIRED)

add_compile_options(-Wall -Wextra -Wpedantic -Werror)

# internal lib
add_library(microgpt_engine src/Value.cpp src/Helpers.cpp src/Model.cpp)

target_include_directories(microgpt_engine PUBLIC include)
target_link_libraries(microgpt_engine PUBLIC fmt::fmt)

# main executable 
add_executable(${PROJECT_NAME} src/main.cpp)
target_link_libraries(${PROJECT_NAME} PRIVATE microgpt_engine)

# tests
enable_testing()
add_executable(microgpt_tests tests/test_value.cpp tests/test_model.cpp)
target_compile_options(microgpt_tests PRIVATE -Wno-c2y-extensions)
target_link_libraries(microgpt_tests PRIVATE
    microgpt_engine
    Catch2::Catch2WithMain
)

include(Catch)
catch_discover_tests(microgpt_tests)
